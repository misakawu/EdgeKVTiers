## 5.1 对应伪代码

### 异常会话筛选

```python
def detect_anomalies(traces, p95_threshold, mig_threshold, qloss_threshold):
    cases = []
    for trace_path in traces:
        reqs = load_trace_events(trace_path)
        if p95(reqs, key="t_ttft_ms") > p95_threshold:
            cases.append(("high_p95", trace_path))
        if count_migrations(reqs) > mig_threshold:
            cases.append(("high_migration", trace_path))
        if cumulative_qloss(reqs) > qloss_threshold:
            cases.append(("quality_over_budget", trace_path))
    return cases
```

### 根因分类

```python
def classify_case(events):
    if oscillates(events, field="q_after"):
        return "degrade_bounce"
    if near_phase_boundary(events) and frequent_q_switch(events):
        return "q_star_jitter"
    if mean_restore_cost(events) > mean_recompute_cost(events):
        return "io_backfire"
    if reuse_pred_error(events) > 0.3:
        return "profile_mismatch"
    if cumulative_qloss(events) > epsilon_budget(events):
        return "epsilon_exhausted"
    return "other"
```

### 失败案例报告生成

```python
def build_failure_table(cases):
    table = []
    for tag, trace_path in cases:
        events = load_trace_events(trace_path)
        root = classify_case(events)
        table.append({
            "case_id": trace_path.stem,
            "symptom": tag,
            "root_cause": root,
            "impact": summarize_impact(events),
            "fix": suggest_fix(root),
        })
    return table
```
