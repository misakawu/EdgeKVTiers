```python
def run_phase_surface(objects, bw_grid, eps_grid):
    rows = []
    for BW in bw_grid:
        for epsilon in eps_grid:
            q_offline = offline_best_tier(objects, BW=BW, epsilon=epsilon)
            q_pred = theorem1_predict(objects, BW=BW, epsilon=epsilon)
            q_online = run_online_tms(objects, BW=BW, epsilon=epsilon)
            rows.append({
                "BW": BW,
                "epsilon": epsilon,
                "q_offline": q_offline,
                "q_pred": q_pred,
                "q_online": q_online,
                "kendall_tau": kendall_tau(q_offline, q_pred),
                "online_match": match_ratio(q_online, q_offline),
            })
    return rows
```

### 定理 2：regret 曲线验证

```python
def run_regret_curve(T_values, trace_generator):
    curve = []
    for T in T_values:
        regrets = []
        for seed in range(10):
            trace = trace_generator(T=T, seed=seed)
            online_cost = run_tms_online(trace)
            offline_cost = run_pareto_opt(trace)
            regrets.append((online_cost - offline_cost) / max_single_loss(trace))
        curve.append({"T": T, "regret_mean": mean(regrets), "regret_std": std(regrets)})
    return curve
```

### 斜率拟合与门槛检查

```python
def fit_regret_slope(curve):
    xs = [log(row["T"]) for row in curve]
    ys = [log(row["regret_mean"]) for row in curve]
    return linear_regression(xs, ys).slope
```
